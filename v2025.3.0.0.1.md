#### Cambio tipado YUP (solo si se actualiza YUP)

En caso retorne error de tipado por validación, se debe aplicar lo siguiente

```typescript
// SI TIENE ESTA FORMA, NO REQUIERE MODIFRICACION
const form = useForm<CuentaPagarInputProp>({
    defaultValues: CUENTA_PAGAR_DEFAULT,
    resolver: yupResolver(cuentaPagarInputValidator),
  })
export const cuentaPagarInputValidator: ObjectSchema<any> = object({
  entrada: mixed().required(),
  acreedor: string().trim().required(),
  fechaInicial: date().required(),
  fechaVencimiento: date().required(),
  descripcion: string().required(),
})

// EN ESTE OTRO CASO se agrega <any, any, any> siempre y cuando cuentaPagarInputValidator no tenga validación
const form = useForm<CuentaPagarInputProp>({
    defaultValues: CUENTA_PAGAR_DEFAULT,
    resolver: yupResolver<any, any, any>(cuentaPagarInputValidator),
  })
export const cuentaPagarInputValidator = object({
  entrada: mixed().required(),
  acreedor: string().trim().required(),
  fechaInicial: date().required(),
  fechaVencimiento: date().required(),
  descripcion: string().required(),
})
```

// Asocia Automaticamente un lote, default false
autoLote?: boolean // Solo cuando esta gestionado por lotes.
// En caso se requiera asociar un lote especifico, default null
codigoLote?: string // Solo cuando esta gestionado por lotes.

// Retorna
return {
